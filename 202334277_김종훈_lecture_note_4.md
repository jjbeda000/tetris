def create_markdown_table(data):
    # 표 머릿글 생성
    table = "| 명령어 | 설명 |\n| ------- | ---- |\n"

    # 데이터 행 추가
    for row in data:
        table += f"| {row[0]} | {row[1]} |\n"

    return table
